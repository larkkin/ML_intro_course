def draw(predictions, svc_object, y, pluses, minuses, X):
    wrong_pluses = X[(y == 1) * (y != predictions)]
    wrong_minuses = X[(y == -1) * (y != predictions)]
    support_indices = set(svc_object.support_)
    support_pluses = X[map(lambda i : i in support_indices and y[i] == 1, range(len(y)))] 
    support_minuses = X[map(lambda i : i in support_indices and y[i] == -1, range(len(y)))] 
    xgrid = ygrid = np.linspace(-2.0, 7, 100)
    Xgrid, Ygrid = np.meshgrid(xgrid, ygrid)
    Z = np.zeros((100, 100))
    for i in range(100):
        for j in range(100):
            Z[i,j] = svc_object.decision_function(((Xgrid[i,j], Ygrid[i,j]),))
    plt.contour(Xgrid, Ygrid, Z, levels=[0])
    plt.plot(pluses["x"], pluses["y"], linestyle="None", marker="+", color="pink", markersize=7)
    #plt.plot(wrong_pluses["x"], wrong_pluses["y"], linestyle="None", marker="+", color="red", markersize=10)
    plt.plot(support_pluses["x"], support_pluses["y"], linestyle="None", marker="+", color="red", markersize=10)
    plt.plot(minuses["x"], minuses["y"], linestyle="None", marker="_", color="cyan", markersize=7)
    #plt.plot(wrong_minuses["x"], wrong_minuses["y"], linestyle="None", marker="_", color="blue", markersize=10)
    plt.plot(support_minuses["x"], support_minuses["y"], linestyle="None", marker="_", color="blue", markersize=10)
    kernel = svc_object.get_params()['kernel']
    plt.title("{} {}".format(kernel, svc_object.get_params()['degree'] if kernel == 'poly' else ''))
    plt.show()
    plt.cla()

def make_svm(kernel, degree, C, gamma, coef0, x_train, x_test, y_train, y_test):
    svm_object = SVC(kernel=kernel, degree=degree, gamma=gamma, coef0=coef0, max_iter=1000)
    svm_object.fit(x_train, y_train)
    prediction_train = svm_object.predict(x_train)
    prediction_test = svm_object.predict(x_test)
    accuracy_train = sum(prediction_train == y_train)*1.0/len(y_train)
    accuracy_test = sum(prediction_test == y_test)*1.0/len(y_test)
    support = len(svm_object.support_)*1.0/len(x_train)
    return (1-accuracy_train, 1-accuracy_test, support)

def draw_plots_for_dataset(dataset_name):
    data = pd.read_csv(dataset_name)
    labels = list(set(data["label"]))
    if len(labels) != 2:
        raise Exception
    data["label"] = np.array([1 if ans == labels[0] else -1 for ans in data["label"]])
    x_train, x_test = train_test_split(data, test_size=0.1)
    y_test = x_test["label"]
    y_train = x_train["label"]
    x_train = x_train.drop("label", axis=1)
    x_test = x_test.drop("label", axis=1)
    kernel = 'poly'
#     gammas = [0.5 / x_train.shape[0], 1.0 / x_train.shape[0], 2.0 / x_train.shape[0], 10.0 / x_train.shape[0]]
    gammas = np.array([0.01, 0.02, 0.05, 0.1, 0.2, 0.5, 1.0, 2.0, 10.0, 100.0]) / x_train.shape[0]
    Cs = [0.001, 0.005, 0.01, 0.05, 0.1, 1.0, 10.0, 100.0]
    coef0s = [-10, 1, 0, 1, 10]
    predictions_poly = []
    for degree in [2,3,5,10]:
        for C in Cs:
            for coef0 in coef0s:
                for gamma in gammas:
                    error_train, error_test, support = make_svm(kernel=kernel, 
                        degree=degree, C=C, gamma=gamma, coef0=coef0, x_train=x_train, x_test=x_test, y_train=y_train, y_test=y_test)
                    predictions_poly.append((error_train, error_test, support))
    errors_train_poly = np.array([p[0] for p in predictions_poly])
    errors_test_poly = np.array([p[1] for p in predictions_poly])
    support_poly = np.array([p[2] for p in predictions_poly])

    kernel = 'rbf'
    predictions_rbf = []
    for C in Cs:
        for gamma in gammas:
            error_train, error_test, support = make_svm(kernel=kernel, 
                degree=degree, C=C, gamma=gamma, coef0=coef0, x_train=x_train, x_test=x_test, y_train=y_train, y_test=y_test)
            predictions_rbf.append((error_train, error_test, support))
    errors_train_rbf = np.array([p[0] for p in predictions_rbf])
    errors_test_rbf = np.array([p[1] for p in predictions_rbf])
    support_rbf = np.array([p[2] for p in predictions_rbf])

    kernel = 'linear'
    predictions_linear = []
    for C in Cs:
        error_train, error_test, support = make_svm(kernel=kernel,
             degree=degree, C=C, gamma=gamma, coef0=coef0, x_train=x_train, x_test=x_test, y_train=y_train, y_test=y_test)
        predictions_linear.append((error_train, error_test, support))
    errors_train_linear = np.array([p[0] for p in predictions_linear])
    errors_test_linear = np.array([p[1] for p in predictions_linear])
    support_linear = np.array([p[2] for p in predictions_linear])

    kernel = 'sigmoid'
    predictions_sigmoid = []
    for C in Cs:
        for coef0 in coef0s:
            for gamma in gammas:
                error_train, error_test, support = make_svm(kernel=kernel, 
                    degree=degree, C=C, gamma=gamma, coef0=coef0, x_train=x_train, x_test=x_test, y_train=y_train, y_test=y_test)
                predictions_sigmoid.append((error_train, error_test, support))
    errors_train_sigmoid = np.array([p[0] for p in predictions_sigmoid])
    errors_test_sigmoid = np.array([p[1] for p in predictions_sigmoid])
    support_sigmoid = np.array([p[2] for p in predictions_sigmoid])

    #support VS train:
    plt.plot(support_poly, errors_train_poly, linestyle="None", marker="o", color="blue", markersize=7)
    plt.plot(support_rbf, errors_train_rbf, linestyle="None", marker="+", color="red", markersize=7)
    plt.plot(support_linear, errors_train_linear, linestyle="None", marker="^", color="orange", markersize=7)
    plt.plot(support_sigmoid, errors_train_sigmoid, linestyle="None", marker="s", color="black", markersize=6)

    plt.xlim((-0.1, 1.1))
    plt.ylim((-0.1, 1.1))
    plt.title("%support vs train error, {}".format(dataset_name.strip(".csv")))
    plt.show()
    plt.cla()

    #support VS test:
    plt.plot(support_poly, errors_test_poly, linestyle="None", marker="o", color="blue", markersize=7)
    plt.plot(support_rbf, errors_test_rbf, linestyle="None", marker="+", color="red", markersize=7)
    plt.plot(support_linear, errors_test_linear, linestyle="None", marker="^", color="orange", markersize=7)
    plt.plot(support_sigmoid, errors_test_sigmoid, linestyle="None", marker="s", color="black", markersize=6)

    plt.xlim((-0.1, 1.1))
    plt.ylim((-0.1, 1.1))
    plt.title("%support vs test error, {}".format(dataset_name.strip(".csv")))
    plt.show()
    plt.cla()


    #support VS test - train:
    plt.plot(support_poly, errors_test_poly - errors_train_poly, linestyle="None", marker="o", color="blue", markersize=7)
    plt.plot(support_rbf, errors_test_rbf - errors_train_rbf, linestyle="None", marker="+", color="red", markersize=7)
    plt.plot(support_linear, errors_test_linear - errors_train_linear, linestyle="None", marker="^", color="orange", markersize=7)
    plt.plot(support_sigmoid, errors_test_sigmoid - errors_train_sigmoid, linestyle="None", marker="s", color="black", markersize=6)

    plt.xlim((-0.1, 1.1))
    plt.ylim((-0.1, 1.1))
    plt.title("%support vs train error - test error, {}".format(dataset_name.strip(".csv")))
    plt.show()
    plt.cla()
