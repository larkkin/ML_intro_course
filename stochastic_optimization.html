def z(x,y):
        return 20*exp(-((x-2.5)**2 + (y+1.5)**2)/5) + \
               3*(1 + (sin(0.08*pi*x*y))**2) +\
               3*log(2 + (10*(x-2.4))**4 + (10*(y+1.3))**4) / (1 + ((x*x + y*y)/100)**(0.1))

    top_perc = generation_size/50

    means = [Xstart,Ystart]

    return generation_scores[-1], means, cov_matrix

def dist(p1, p2):
    return abs(p2[1]-p1[1]) + abs(p2[0]-p1[0])
def path_length(points):
    path_len = 0
    for i in xrange(len(points)-1):
        p1 = points[i]
        p2 = points[i+1]
        path_len += dist(p1, p2)
    return path_len

def delta_len(i, j, points):
    n = len(points)
    res = 0

    if i > 0:
        res += dist(points[i-1],points[j]) - dist(points[i-1],points[i])
    if i < j - 1 :
        res += dist(points[i+1],points[j]) - dist(points[i+1],points[i])
        res += dist(points[j-1],points[i]) - dist(points[j-1],points[j])
    if j < n-1:
        res += dist(points[j+1],points[i]) - dist(points[j+1],points[j])
    
    return res

def hill_climbing(points, num_iterations, num_variants):
    n = len(points)
    best_i = -1
    best_j = -1
    best_delta = 0
    for iteration in xrange(num_iterations):
        old_best_i = best_i
        old_best_j = best_j
        old_best_delta = best_delta
        best_i = -1
        best_j = -1
        best_delta = 0
        for variant in xrange(num_variants):
            i = randint(0, n-1)
            j = randint(0, n-1)
            j, i = max(i,j), min(i,j)
            if i == j:
                continue
            delta = delta_len(i,j,points)
            if delta < best_delta:
                best_i = i
                best_j = j
                best_delta = delta
        if best_delta < 0:
            points[best_i], points[best_j] = points[best_j], points[best_i]
        else:
            best_i = old_best_i
            best_j = old_best_j
            best_delta = old_best_delta

def harmonic():
    i = 1
    q = 1
    while q > 0:
        yield q
        i += 1
        q = 1.0/i
    while True:
        yield 0.00000001

def crossingover(points1, points2):
    n = len(points1)
    start = randint(0,n/2)
    res_points = points1[start:start+n/2]
    res_set = set(res_points)
    res_points.extend(p for p in points2 if p not in res_set)
    return res_points

def GA(points, generation_size, proceeding_size, proceeding_prob, num_iterations):
    generation = []
    for i in xrange(generation_size):
        creature = points[::]
        shuffle(creature)
        generation.append((path_length(creature),creature))

    for iteration in xrange(num_iterations):
        # extinction
        generation.sort(key=lambda x: x[0])
        proceeding = [[] for i in xrange(proceeding_size)]
        proceeding[0] = generation[0][1]
        proceeded = 1
        i = 0
        while proceeded < proceeding_size and i < proceeding_size:
            p = random()
            if p < proceeding_prob:
                proceeding[proceeded] = generation[i][1]
                proceeded += 1
            i += 1
        if proceeded < 2: 
            proceeding.append(choice(generation[1:])[1]) 
        generation = [p for p in proceeding if p != []]
        # breeding
        current_generation_size = len(generation)
        while len(generation) < generation_size:
            parent_1 = choice(generation[:current_generation_size])
            parent_2 = choice(generation[:current_generation_size])
            creature = crossingover(parent_1,parent_2)
            random_transpose(creature)
            generation.append(creature)
        for i in xrange(generation_size):
            generation[i] = (path_length(generation[i]), generation[i])

    for i in xrange(len(points)):
        points[i] = generation[0][1][i]
