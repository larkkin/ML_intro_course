def perceptron(X, y, iterations_number):
    w = [0.0]
    for col in X: #number of columns - coordinates
        if col == "intercept":
            continue
        min_in_the_column = X[col].min()
        #print min_in_the_column
        max_in_the_column = X[col].max()
        #print max_in_the_column
        w.append(uniform(min_in_the_column, max_in_the_column))
    w = np.array(w)

    number_of_points = X.shape[0]
    best_result = [] # 1 is for line coordinates, 2 is for accuracy
    best_result.append(w)
    best_result.append(0)
    points_below_line = [0, 0, 0]
    points_above_line = [0, 0, 0]
    i = 0
    while i < iterations_number:
        predictions = np.array([1 if (row * w).sum() > 0 else -1 for rowindex, row in X.iterrows()])
        accuracy = np.sum(predictions == y) * 1.0 / number_of_points
        if accuracy > best_result[1]:
            best_result[0] = np.array(w)
            best_result[1] = accuracy
        # print accuracy
        if accuracy == 1.0:
            break
        misclassified_points_indices = np.flatnonzero(predictions != y)
        misclassified_point_index = choice(misclassified_points_indices) 
        misclassified_point = X.iloc[misclassified_point_index]
        w += misclassified_point * y[misclassified_point_index]
        i += 1
    return best_result
